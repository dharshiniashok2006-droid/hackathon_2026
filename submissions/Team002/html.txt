    <input type="number" id="assignment" placeholder="Assignment Score" required>
    <input type="number" id="study_hours" placeholder="Study Hours / Day" required>
    <button type="submit">Predict</button>
</form>

<h3 id="resultMarks"></h3>
<h3 id="resultStatus"></h3>

<script>
document.getElementById("predictionForm").addEventListener("submit", function(e){
    e.preventDefault(); 

    
    let attendance = parseFloat(document.getElementById("attendance").value);
    let internal = parseFloat(document.getElementById("internal").value);
    let assignment = parseFloat(document.getElementById("assignment").value);
    let study_hours = parseFloat(document.getElementById("study_hours").value);

    
    let prediction = attendance*0.2 + internal*0.3 + assignment*0.2 + study_hours*3;

    
    let status = prediction >= 50 ? "PASS " : "FAIL ";


    document.getElementById("resultMarks").innerText = "Predicted Marks: " + prediction.toFixed(2);
    document.getElementById("resultStatus").innerText = "Status: " + status;
});
</script>
